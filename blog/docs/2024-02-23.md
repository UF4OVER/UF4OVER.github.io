        if filename.endswith('.md'):
            # 获取文件信息
            file_path = os.path.join(docs_path, filename)
            file_stat = os.stat(file_path)
            
            # 获取文件修改时间
            mod_time = datetime.fromtimestamp(file_stat.st_mtime)
            date_str = mod_time.strftime('%Y-%m-%d')
            
            # 读取文件内容，提取标签
            tags = []
            with open(file_path, 'r', encoding='utf-8') as f:
                first_line = f.readline().strip()
                if first_line.startswith('tags:'):
                    try:
                        tags = eval(first_line[5:].strip())
                    except:
                        pass
            
            # 使用相对路径
            relative_path = './docs/' + filename
            
            article = {
                'title': filename,
                'date': date_str,
                'path': relative_path,
                'tags': tags
            }
            
            articles.append(article)
    
    # 按日期排序
    articles.sort(key=lambda x: x['date'], reverse=True)
    
    # 生成 JavaScript 配置文件
    config = f'const BLOG_ARTICLES = {json.dumps(articles, indent=2, ensure_ascii=False)};'
    
    # 写入配置文件
    config_path = os.path.abspath('../blog/articles-config.js')
    with open(config_path, 'w', encoding='utf-8') as f:
        f.write(config)
    
    print(f'找到 {len(articles)} 篇文章')
    print('配置文件已生成')
